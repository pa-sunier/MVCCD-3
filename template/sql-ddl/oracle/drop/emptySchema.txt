--Supprimer les tables
--Lancer cette requête, copier le résultat et l'exécuter ensuite
select 'drop table ' || object_name || ' cascade constraints purge;'
from user_objects
where object_type = 'TABLE';

--Supprimer autres objets
--Lancer cette requête, copier le résultat et l'exécuter ensuite
select 'drop ' || object_type || ' ' || object_name || ';'
from user_objects
where object_type IN ('SEQUENCE', 'TRIGGER', 'PACKAGE',
'FUNCTION', 'PROCEDURE', 'VIEW', 'DATABASE LINK', 'SYNONYM');

--Purger la poubelle
purge recyclebin;

--contrôle
select count(*) from user_objects;
