CREATE OR REPLACE TRIGGER {triggerName}
BEFORE INSERT ON {tableName} FOR EACH ROW
DECLARE
    {vl_newrec} {tableName}%ROWTYPE;

BEGIN
    {-NODML-}
    {-TRIGGER_NEW_TO_RECORD-}
    {packageName}.{autogen_columns_ins}(vl_newrec);
    {packageName}.{autogen_columns}(vl_newrec);
    {packageName}.{checktype_columns}({vl_newrec});
    {packageName}.{uppercase_columns}(vl_newrec);
    {packageName}.{column_PEA}({vl_newrec});
    {-INS_JN_INS-}
    {-TRIGGER_RECORD_TO_NEW-}
END;
